Example init.json

{
  "url": "127.0.0.1/fotobox/index.html",
  "defaultUrl": "http://127.0.0.1",
  "defaultPort": 8080,
  "localIndexPath": "wwwroot/index.html",
  "kiosk": false,
  "title": "Fotobox Kiosk",
  "icon": "Assets/app.ico",
  "minimizeToTray": true,
  "allowDevTools": false
}

JavaScript-Beispiele auf Deutsch

1) Minimieren

if (window.hostApp) {
  window.hostApp.minimize();
}

2) Kioskmodus aktivieren

if (window.hostApp) {
  window.hostApp.maximize();
}

// oder explizit
if (window.hostApp) {
  window.hostApp.setKiosk(true);
}

3) Kioskmodus verlassen und Fenster normal anzeigen

if (window.hostApp) {
  window.hostApp.restore();
}

// oder explizit
if (window.hostApp) {
  window.hostApp.setKiosk(false);
}

4) Programm beenden

if (window.hostApp) {
  window.hostApp.exit();
}

// alias
if (window.hostApp) {
  window.hostApp.close();
}

5) Button direkt verbinden

<button id="exitButton">Beenden</button>
<script>
  document.getElementById('exitButton').addEventListener('click', () => {
    if (window.hostApp) {
      window.hostApp.exit();
    }
  });
</script>

JavaScript examples in English

1) Minimize

if (window.hostApp) {
  window.hostApp.minimize();
}

2) Enter kiosk mode

if (window.hostApp) {
  window.hostApp.maximize();
}

// or explicit
if (window.hostApp) {
  window.hostApp.setKiosk(true);
}

3) Leave kiosk mode and show normal window

if (window.hostApp) {
  window.hostApp.restore();
}

// or explicit
if (window.hostApp) {
  window.hostApp.setKiosk(false);
}

4) Exit application

if (window.hostApp) {
  window.hostApp.exit();
}

// alias
if (window.hostApp) {
  window.hostApp.close();
}

5) Wire a button directly

<button id="exitButton">Exit</button>
<script>
  document.getElementById('exitButton').addEventListener('click', () => {
    if (window.hostApp) {
      window.hostApp.exit();
    }
  });
</script>
